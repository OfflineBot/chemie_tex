\section{Homologe Reihen}
Beispiele sind jeweils mit Ethan- gewählt.
\begin{enumerate}
    \item Alkane: $C-C-$Einfachbindung $|\ C_nO_{n2 + 2}$ \\
        Ethan: 

        \scalebox{0.8}{\chemfig{H-C(-[2]H)(-[6]H)-C(-[2]H)(-[6]H)-H}}

    \item Alkene: $C=C-$Doppelbindung $|\ C_nO_{n2}$ \\
        Ethen:

        \scalebox{0.8}{\chemfig{H-C(-[2]H)(-[6]H)=C(-[2]H)(-[6]H)-H}}

    \item Alkine: $C\equiv C-$Dreifachbindung $|\ C_nO_{n2 - 2}$ \\
        Ethin: 

        \scalebox{0.8}{\chemfig{H-C(-[2]H)(-[6]H)~C(-[2]H)(-[6]H)-H}}

    \item Alkohole: $R-OH$ $|\ C_nH{2n+1}OH$\\
        Ethanol:

        \scalebox{0.8}{\chemfig{C(-[2]OH)(-[4]H)(-[6]H)-C(-[2]H)(-[6]H)-H}}

    \item Aldehyde: $R-CHO$ $|\ C_nH_{2n}CHO$ \\
        Ethanal: 

        \scalebox{0.8}{\chemfig{C(=[3]O)(-[5]H)-C(-[2]H)(-[6]H)-H}}

    \item Ketone: $R-CO-R'$ $|\ C_nH_{2n}O$ \\
        Propanon: (Da mindestens 3 $C$ benötigt)

        \scalebox{0.8}{\chemfig{
            H-C(-[2]H)(-[6]H)
            -C(=[2]O)
            -C(-[2]H)(-[6]H)-H
        }}

    \item Carbonsäuren: $R-COOH$ $|\ C_nH_{2n+1}COOH$ \\
        Ethansäure: 

        \scalebox{0.8}{\chemfig{C(=[3]O)(-[5]O(-[5]H))-C(-[2]H)(-[6]H)-H}}

    \item Ether: $R-COO-R'$ \\
        Diethylether-Molekül:

        \scalebox{0.8}{\chemfig{H-C(-[2]H)(-[6]H)-C(-[2]H)(-[6]H)-O-C(-[2]H)(-[6]H)-C(-[2]H)(-[6]H)-H}}

    \item Ester: $R-COO-R'$ \\
        Ethansäureethylester:

        \scalebox{0.8}{\chemfig{H-C(-[2]H)(-[6]H)-C(=[2]O)-O-C(-[2]H)(-[6]H)-C(-[2]H)(-[6]H)-H}}

\end{enumerate}
